017package net.ljcomputing.randy.randomizer;
018
019import java.io.IOException;
020import java.io.InputStream;
021import java.lang.reflect.InvocationTargetException;
022import java.util.Properties;
023
024import net.ljcomputing.randy.data.DataSource;
025import net.ljcomputing.randy.data.file.DataSourceType;
026import net.ljcomputing.randy.exception.GeneratorException;
027import net.ljcomputing.randy.exception.RandomizerException;
028import net.ljcomputing.randy.generator.BaseGenerator;
029import net.ljcomputing.randy.generator.Generator;
030
031/**
032 * @author James G. Willmore
033 *
034 */
035public abstract class AbstractRandomizer { //NOPMD
036
037  /** The Constant PROPERTIES_FILENAME. */
038  private static final String PROP_FILENAME = "application.properties";
039
040  /** The Constant PROPERTY_URL. */
041  private static final String PROP_URL = ".url";
042
043  /** The Constant PROPERTY_BUFFER_SIZE. */
044  private static final String PROP_BUFFER_SIZE = ".bufferSize";
045
046  /** The properties. */
047  protected static final transient Properties PROPERTIES = new Properties();
048
049  /** The generator. */
050  protected final transient Generator generator;
051
052  /**
053   * Instantiates a new abstract randomizer.
054   *
055   * @param propertiesPrefix the properties prefix
056   * @throws RandomizerException the randomizer exception
057   */
058  public AbstractRandomizer(final String propertiesPrefix) throws RandomizerException {
059    initProperties();
060
061    try {
062      final String url = PROPERTIES.getProperty(propertiesPrefix + PROP_URL);
063      final String bufferSizeProp = PROPERTIES.getProperty(propertiesPrefix + PROP_BUFFER_SIZE);
064      final Integer bufferSize = Integer.parseInt(bufferSizeProp);
065      final DataSourceType dsType = DataSourceType.getType(url);
066      final DataSource dataSource = dsType.getDataSourceImpl().getConstructor(String.class)
067          .newInstance(url);
068      this.generator = new BaseGenerator(dataSource, bufferSize);
069    } catch (GeneratorException | InstantiationException | IllegalAccessException
070        | IllegalArgumentException | InvocationTargetException | NoSuchMethodException
071        | SecurityException exception) {
072      throw new RandomizerException("Exception encountered instaniating randomizer", exception);
073    }
074  }
075
076  /**
077   * Inits the properties.
078   *
079   * @throws RandomizerException the randomizer exception
080   */
081  private static void initProperties() throws RandomizerException {
082    final Thread thread = Thread.currentThread(); //NOPMD
083    final ClassLoader classLoader = thread.getContextClassLoader(); //NOPMD
084    final InputStream inStream = classLoader.getResourceAsStream(PROP_FILENAME); //NOPMD
085
086    try {
087      PROPERTIES.load(inStream);
088    } catch (IOException exception) {
089      throw new RandomizerException("Randomizer exception", exception);
090    }
091  }
092}
