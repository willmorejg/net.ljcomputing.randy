017package net.ljcomputing.randy.data.file;
018
019import java.io.IOException;
020import java.io.Reader;
021import java.io.StringReader;
022import java.net.URI;
023import java.nio.file.Files;
024import java.nio.file.Path;
025import java.nio.file.Paths;
026import java.util.List;
027import java.util.NoSuchElementException;
028import java.util.stream.Stream;
029
030import org.apache.commons.csv.CSVFormat;
031import org.apache.commons.csv.CSVParser;
032import org.apache.commons.csv.CSVRecord;
033
034import net.ljcomputing.randy.data.DataSource;
035import net.ljcomputing.randy.exception.DataSourceException;
036
037/**
038 * The CSV data source.
039 * 
040 * @author James G. Willmore
041 *
042 */
043public class CsvDataSource extends AbstractFileDataSource implements DataSource {
044
045  /** The max file size. */
046  private transient long maxFileSize;
047
048  /**
049   * Instantiates a new file data source.
050   *
051   * @param uri the uri
052   * @throws DataSourceException the data source exception
053   */
054  public CsvDataSource(final String uri) throws DataSourceException {
055    super(uri);
056    countLines();
057  }
058
059  /**
060   * Convert the URI to determine the underlying data source 
061   * (the file to use as the CSV data source).
062   *
063   * @return the string
064   */
065  private String convertUri() {
066    final StringBuilder result = new StringBuilder();
067
068    if (getDataSource() != null) {
069      final URI theUri = getDataSource();
070      final String rawUri = theUri.toString(); //NOPMD
071      final int colonIdx = rawUri.indexOf(':'); //NOPMD
072      result.append(rawUri.substring(colonIdx + 1)); //NOPMD
073    }
074
075    return result.toString();
076  }
077
078  /**
079   * Gets the stream.
080   *
081   * @return the stream
082   * @throws IOException Signals that an I/O exception has occurred.
083   * @throws DataSourceException the data source exception
084   */
085  private Stream<String> getStream() throws IOException, DataSourceException {
086    final URI newUri = URI.create(convertUri());
087    final Path path = Paths.get(newUri);
088    return Files.lines(path);
089  }
090
091  /**
092   * Count lines.
093   *
094   * @throws DataSourceException the data source exception
095   */
096  private void countLines() throws DataSourceException {
097    try {
098      final Stream<String> lines = getStream();
099      maxFileSize = (int) lines.count(); //NOPMD 
100      lines.close(); //NOPMD
101    } catch (IOException exception) {
102      throw new DataSourceException("IO Exception", exception);
103    } catch (NoSuchElementException exception) {
104      throw new DataSourceException("Given record does not exist.", exception);
105    }
106  }
107
108  /**
109   * @see net.ljcomputing.randy.data.DataSource#read(int)
110   */
111  @Override
112  public String read(final int record) throws DataSourceException {
113    String result;
114
115    try {
116      final String value = readRecordValue(record);
117      final StringReader reader = new StringReader(value);
118      result = getValue(reader, 0); //NOPMD
119      reader.close(); //NOPMD
120    } catch (IOException exception) {
121      throw new DataSourceException("IO Exception", exception);
122    } catch (NoSuchElementException exception) {
123      throw new DataSourceException("Record " + record + " does not exist.", exception);
124    }
125
126    return result;
127  }
128  
129  /**
130   * Read record value.
131   *
132   * @param record the record
133   * @return the string
134   * @throws DataSourceException the data source exception
135   */
136  private String readRecordValue(final int record) throws DataSourceException {
137    try {
138      final Stream<String> lines = getStream();
139      final String line = lines.skip(record).findFirst().get(); //NOPMD
140      lines.close(); //NOPMD
141      return line;
142    } catch (IOException exception) {
143      throw new DataSourceException("IO Exception", exception);
144    }
145  }
146  
147  /**
148   * Gets the csv parser.
149   *
150   * @param reader the reader
151   * @param index the index
152   * @return the csv parser
153   * @throws IOException IOException
154   */
155  private static final String getValue(final Reader reader, final int index) throws IOException {
156    @SuppressWarnings("resource")
157    final CSVParser csvParser = new CSVParser(reader, CSVFormat.RFC4180);
158    final List<CSVRecord> csvRecords = csvParser.getRecords();
159    final CSVRecord csvRecord  = csvRecords.get(0); //NOPMD
160    return csvRecord.get(index); //NOPMD
161  }
162
163  /**
164   * @see net.ljcomputing.randy.data.file.AbstractFileDataSource#getMaxSize()
165   */
166  @Override
167  public long getMaxSize() {
168    return maxFileSize;
169  }
170}
